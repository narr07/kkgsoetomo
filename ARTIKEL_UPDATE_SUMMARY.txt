🔍 QUERIES USED
─────────────────────────────────────────────────────────────

import { allArticlesQuery, articleBySlugQuery, relatedArticlesQuery } from '@/sanity/lib/queries'

allArticlesQuery
  → Fetch all articles with author, category, image
  
articleBySlugQuery (+ slug param)
  → Fetch single article with full content
  
relatedArticlesQuery (+ categoryId, currentId params)
  → Fetch related articles from same category
